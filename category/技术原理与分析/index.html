import (
	"fmt"
	"net"
	"os"
)
func main() {
	if len(os.Args) != 2 {
		fmt.Fprintf(os.Stderr, "Usage: %s ip.addr\n", os.Args[0])
		os.Exit(1)
	}
	addr := os.Args[1]
	ip := net.ParseIP(addr)
	if ip != nil {
		fmt.Println("The address is", ip.String())
	} else {
		fmt.Println("Invalid address")
	}
	os.Exit(0)
}
//ParseIP函数能够解析正确的10进制IPv4和IPv6地址，以下为函数定义和测试
func ParseIP(s string) IP {
	for i := 0; i < len(s); i++ {
		switch s[i] {
		case '.':
			return parseIPv4(s)
		case ':':
			return parseIPv6(s)
		}
	}
	return nil
}

反弹shell原理及分析

反向shell工作原理

为什么需要反向shell

为了控制目标系统，我们都会尝试去获取交互式shell，而大部分的系统都处在防火墙之后，大部分服务器只接受指定端口的连接，例如web服务器只接受80端口和443端口的连接，我们想要正向直接获取shell是不太可能的事情，但大部分防火墙并不限制传出连接，因此我们需要反向shell

工作原理

反向shell与正向shell相反，监控端监听某端口的tcp/UDP连接，目标主机发送连接请求到该端口，并将其输入输出转换到监控端，实现了角色的反转

系统提权

提权基础

权限提升在渗透测试中是必不可少的环节，为了得到更高的权限，攻击者可以利用各式各样的工具，漏洞来进行权限的提升

用户权限

在Linux中一个文件有3种权限。对文件而言用户有3种不同类型：文件所有者、群组用户、其他用户。例如：chmod 777中，三个数字7分别对应上面三种用户，权限值都为7。

  • 文件权限：
    • r 只读
    • w 只写
    • x 执行

高权限可进一步利用的方法：

  1. 重置其他账户密码访问其他账户权限文件
  2. 绕过访问控制的权限操作数据内容
  3. 更改软件的配置
  4. 实现持久化
  5. 更改对应用户权限
